280inline TimeInterval::DiscreteTimePointRange TimeInterval::discrete(double step) const
281{
282 double dur = duration();
283 if (step <= 0.0 || dur <= 0.0) {
284 return DiscreteTimePointRange(*this, step, 0);
285 }
286 size_t n = static_cast<size_t>(std::ceil(dur / step));
287 return DiscreteTimePointRange(*this, step, n);
288}

290inline TimeInterval::DiscreteEpochSecondRange TimeInterval::discrete(const TimePoint& epoch, double step) const
291{
292 double dur = duration();
293 if (step <= 0.0 || dur <= 0.0) {
294 return DiscreteEpochSecondRange(0.0, step, 0.0, 0);
295 }
296 size_t n = static_cast<size_t>(std::ceil(dur / step));
297 double offset = getStart() - epoch;
298 double stopOffset = getStop() - epoch;
299 return DiscreteEpochSecondRange(offset, step, stopOffset, n);
300}

302inline errc_t TimeInterval::merge(const TimeInterval &other)
303{
304 const TimePoint& thisStart = start();
305 const TimePoint& otherStart = other.start();
306 TimePoint thisStop = stop();
307 TimePoint otherStop = other.stop();
308
309 if (thisStart.durationFrom(otherStop) > 0 || otherStart.durationFrom(thisStop) > 0)
310 {
311 aError("merge time interval failed, no overlap");
312 return eErrorInvalidParam;
313 }
314
315 const TimePoint& mergedStart = (thisStart.durationFrom(otherStart) <= 0) ? thisStart : otherStart;
316 const TimePoint& mergedStop = (thisStop.durationFrom(otherStop) >= 0) ? thisStop : otherStop;
317
318 setStartStop(mergedStart, mergedStop);
319 return eNoError;
320}
