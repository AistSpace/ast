158A_ALWAYS_INLINE KinematicTransform KinematicTransform::composed(const KinematicTransform &next) const
159{
160 KinematicRotation rotation = this->getKinematicRotation().composed(next.getKinematicRotation());
161 /*
162 也可以通过 getRotation().transformVectorVelocityInv() 来实现。
163 */
164 Vector3d vector = next.getTranslation() * this->getMatrix();
165
166 Vector3d translation = this->getTranslation() + vector;
167 Vector3d velocity = this->getVelocity() + next.getVelocity() * this->getMatrix()
168 + this->getRotationRate().cross(vector);
169 return KinematicTransform(translation, velocity, rotation);
170}
